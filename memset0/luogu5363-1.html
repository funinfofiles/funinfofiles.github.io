洛谷5363 [SDOI2019]移动金币

  • 2019-05-22
  • 题解

n 个格子上有 m 个棋子，Alice 和 Bob 轮流行动，每次可以选择一个棋子向左移动至少一步，但不能跨越已有的棋子，不能的移动的一方算输，求 Alice 必胜的方案数。

这题需要我们知道一个叫做阶梯 Nim 的东东。假设 m 个棋子的下标分别为 a_{1 \cdots n} ，令 \{b\} 为 \{a\} 的差分数组：b_{i} = a_i - a_{i-1} 。假设给第 k 个棋子移动了 p 步，也就相当于 a[k] -= v ，即 b[k] -= v, b[k+1] += v 。如果把 b 数组反过来并给每一位 -= 1 就相当于一个朴素的阶梯 Nim 。

根据阶梯 Nim 的结论得，我们要给 m 个位置填数，使得其且和小于等于 n ，且下标为奇数的 \lfloor \frac {m+1} 2 \rfloor 个位置的异或和等于 0，求方案数。我们可以算出等于零的方案数在用 \binom n m 减即可得到答案。考虑异或和为 0 这个条件比较好维护，我们可以一对一对的分配二进制位，则：

F_k(x) = \sum_{i=0}^{\lfloor \frac {\lfloor \frac {m + 1} 2 \rfloor} 2 \rfloor} x^{2i \times 2^k} \binom {\lfloor \frac {m+1} 2 \rfloor} {2i} \\ ans = \sum_{i=0}^{n-m} \sum_{j=0}^i \binom {n - m + \lfloor \frac m 2 \rfloor - 1} {\lfloor \frac m 2 \rfloor - 1} [x^j] \prod_{k=0}^{\left \lfloor \log n \right \rfloor - 1} F_k(x)

这题的模数不适合我们卷积，但是可以发现多项式的项数特别少，我们可以直接用 DP 来维护转移，时间复杂度 \mathcal O(n m \log n)。

代码：

// =================================
//   author: memset0
//   date: 2019.05.22 12:38:19
//   website: https://memset0.cn/
// =================================
#include <bits/stdc++.h>
#define ll long long
#define debug(...) ((void)0)
#ifndef debug
#define debug(...) fprintf(stderr,__VA_ARGS__)
#endif
namespace ringo {
template <class T> inline void read(T &x) {
    x = 0; char c = getchar(); bool f = 0;
    while (!isdigit(c)) f ^= c == '-', c = getchar();
    while (isdigit(c)) x = x * 10 + c - '0', c = getchar();
    if (f) x = -x;
}
template <class T> inline void print(T x) {
    if (x < 0) putchar('-'), x = -x;
    if (x > 9) print(x / 10);
    putchar('0' + x % 10);
}
template <class T> inline void print(T x, char c) { print(x), putchar(c); }
template <class T> inline void print(T a, int l, int r, std::string s = "") {
    if (s != "") std::cout << s << ": ";
    for (int i = l; i <= r; i++) print(a[i], " \n"[i == r]);
}

const int N = 2e5 + 10, mod = 1e9 + 9;
int n, m, ans, f[N], g[N], fac[N], ifac[N];

inline int dec(int a, int b) { a -= b; return a < 0 ? a + mod : a; }
inline int inc(int a, int b) { a += b; return a >= mod ? a - mod : a; }
inline int mul(int a, int b) { return (ll)a * b - (ll)a * b / mod * mod; }
inline int inv(int x) { return x < 2 ? 1 : mul(mod - mod / x, inv(mod % x)); }
inline int C(int n, int m) { return n < m ? 0 : mul(fac[n], mul(ifac[m], ifac[n - m])); }

void main() {
    read(n), read(m);
    f[0] = fac[0] = fac[1] = ifac[0] = ifac[1] = 1;
    for (int i = 2; i <= n; i++) fac[i] = mul(fac[i - 1], i);
    for (int i = 2; i <= n; i++) ifac[i] = mul(mod - mod / i, ifac[mod % i]);
    for (int i = 2; i <= n; i++) ifac[i] = mul(ifac[i - 1], ifac[i]);
    for (int k = 0, p = m - (m >> 1); (1 << (k + 1)) <= n - m; k++) {
        for (int i = 0; i <= n - m; i++) g[i] = f[i];
        for (int u = 1, delta; (delta = (u << 1) * (1 << k)) <= n - m && (u << 1) <= p; u++) {
            // printf("k=%d u=%d delta=%d\n", k, u, delta);
            for (int i = n - m; i >= delta; i--)
                f[i] = inc(f[i], mul(g[i - delta], C(p, u << 1)));
        }
    }
    // print(f, 0, n - m, "f");
    for (int i = 1; i <= n - m; i++) f[i] = inc(f[i], f[i - 1]);
    // print(f, 0, n - m, "f");
    for (int i = 0, p = (m >> 1); i <= n - m; i++) {
        ans = inc(ans, mul(f[n - m - i], C(i + p - 1, p - 1)));
        // printf("%d : %d %d\n", i, f[n - m - i], C(i + p - 1, p - 1));
    }
    // printf(">> %d %d\n", C(n, m), ans);
    print(dec(C(n, m), ans), '\n');
}

} signed main() {
#ifdef MEMSET0_LOCAL_ENVIRONMENT
    freopen("1.in", "r", stdin);
#endif
    return ringo::main(), 0;
}
