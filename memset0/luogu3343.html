洛谷3343 [ZJOI2015]地震后的幻想乡

  • 2019-06-22
  • 题解

n 个点之间有 m 条边连接，现在由于地震而断裂了，每条边的恢复时间在 [0, 1] 中随机均匀生成，求使 n 个点联通的期望时间。1 \leq n \leq 10，保证没有重边和自环。

Link

重建所需的时间为 \lambda ，我们需要求 E(\lambda) 。

定义 f(x) 为 \lambda 的概率密度函数，F(x) = P(\lambda \geq x) 。

\begin{aligned} E(\lambda) &= \int_0^1 f(x) x \mathrm dx \\ &= \int_0^1 f(x) \left( \int_0^x 1 \mathrm dt \right) \mathrm dx \\ &= \int_0^1 \int_x^1 f(x) \mathrm dx \mathrm dt \\ &= \int_0^1 F(t) \mathrm dt \end{aligned}

考虑 F(x) = P(\lambda \geq x) 的实际意义，即在 x 时刻图不连通的概率。定义对于一个点集 S ，在时刻 x 仍不连通的概率为 P_S(x) 。显然我们可以枚举一个 S 的真子集 T ，要求 T 与 (S - T) 之间无边且 T 本身联通。定义 T(A, B) 表示两个不重点集 A, B 之间的边数。为了避免重复计算，我们需要保证 1 \in S 且 1 \in T

P_S(x) = \sum_{1 \in T \subset S} (1 - x)^{T(T, S - T)} \left(1 - P_{T}(x)\right) \ \ \ (1 \in S)

边界条件有 P_{\{1\}} = 0 。

也就是说现在我们需要求 \displaystyle E(\lambda) = \int_0^1 P_{[1, n]} (x) \mathrm dx 。

\begin{aligned} E(\lambda) &= \int_0^1 P_{[1, n]} (x) \mathrm dx \\ &= \int_0^1 \sum_{1 \in S \subset [1, n]} (1 - x)^{T(S, [1, n] - S)} \left(1 - P_S(x) \right) \mathrm dx \\ &= \sum_{1 \in S \subset [1, n]} \left( \int_0^1 (1-x)^{T(S, [1, n] - S)} \mathrm dx - \int_0^1 (1-x)^{T(S, [1, n] - S)} P_S(x) \mathrm dx \right) \\ &= \sum_{1 \in S \subset [1, n]} \left( \frac 1 {1 + T(S, [1, n] - S)} - \int_0^1 (1-x)^{T(S, [1, n] - S)} P_S(x) \mathrm dx \right) \\ \end{aligned}

考虑如何求

\begin{aligned} & \int_0^1 (1-x)^k P_S(x) \mathrm dx \ \ \ (1 \in S)\\ =& \int_0^1 (1-x)^k \left( \sum_{1 \in T \subset S} (1-x)^{T(T, S - T)} \left( 1 - P_{T}(x) \right) \right) \mathrm dx \\ =& \sum_{1 \in T \subset S} \int_0^1 (1 - x)^{k + T(T, S - T)} \left( 1 - P_{T}(x) \right) \mathrm dx \\ =& \sum_{1 \in T \subset S} \left( \frac 1 {1 + k + T(T, S - T)} - \int_0^1 (1-x)^{k + T(T, S - T)} P_{T}(x) \mathrm dx \right) \end{aligned}

边界条件为 \displaystyle \int_0^1 (1-x)^k P_{\{1\}}(x) \mathrm dx = 0 。

显然这个可以递推维护，我们枚举集合 S 和整数 k ，总时间复杂度为枚举子集的时间复杂度，即 O \left(3^n (n + m) \right) 。

    x = 0; char c = getchar(); bool f = 0;
    while (!isdigit(c)) f ^= c == '-', c = getchar();
    while (isdigit(c)) x = x * 10 + c - '0', c = getchar();
    if (f) x = -x;
}
template <class T> inline void print(T x) {
    if (x < 0) putchar('-'), x = -x;
    if (x > 9) print(x / 10);
    putchar('0' + x % 10);
}
template <class T> inline void print(T x, char c) { print(x), putchar(c); }
template <class T> inline void print(T a, int l, int r, std::string s = "") {
    if (s != "") std::cout << s << ": ";
    for (int i = l; i <= r; i++) print(a[i], " \n"[i == r]);
}

#define in(k, S) (((S) >> ((k) - 1)) & 1)
const int N = 20, M = 100, P = 1024;
int n, m;
double ans, inv[M], dp[P][M];
std::vector<int> G[N];
std::vector<std::pair<int, int>> E;

void main() {
    read(n), read(m);
    for (int i = 1, u, v; i <= m; i++) {
        read(u), read(v);
        E.push_back(std::make_pair(u, v));
        G[u].push_back(v), G[v].push_back(u);
    }
    for (int i = 0; i <= m; i++) inv[i] = 1 / (double)(i + 1);
    for (int S = 1; S < (1 << n); S += 2) {
        std::vector<int> bit;
        for (int i = 1; i <= n; i++)
            if (in(i, S))
                bit.push_back(i);
        for (int T_ = 1; T_ < (1 << bit.size()) - 1; T_ += 2) {
            int T = 0;
            for (int i = 0; i < bit.size(); i++)
                if (in(i + 1, T_))
                    T |= 1 << (bit[i] - 1);
            int cnt = 0;
            for (auto it : E)
                if (in(it.first, S) && in(it.second, S) && (in(it.first, T) + in(it.second, T) == 1))
                    ++cnt;
            // printf("S ");
            // for (int i = 1; i <= n; i++) print(in(i, S));
            // printf(" | T ");
            // for (int i = 1; i <= n; i++) print(in(i, T));
            // printf(" | T_ ");
            // for (int i = 1; i <= n; i++) print(in(i, T_));
            // printf(" => ");
            // for (int i = 1; i <= n; i++) if (in(i, T)) print(i, ' ');
            // printf("| ");
            // for (int i = 1; i <= n; i++) if (!in(i, T) && in(i, S)) print(i, ' ');
            // printf("=> %d\n", cnt);
            for (int k = 0; k + cnt <= m; k++)
                dp[S][k] += inv[k + cnt] - dp[T][k + cnt];
        }
        if (S != (1 << n) - 1) {
            int cnt = 0;
            for (auto it : E)
                if (in(it.first, S) + in(it.second, S) == 1)
                    ++cnt;
            ans += inv[cnt] - dp[S][cnt];
        }
    }
    // for (int S = 0; S < (1 << n); S++) {

    //     printf(" : ");
    //     for (int i = 0; i <= m; i++)
    //         printf("%.3lf%c", dp[S][i], " \n"[i == m]);
    // }
    printf("%.6lf\n", ans);
}
