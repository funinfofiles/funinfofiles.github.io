洛谷3779 [SDOI2017]龙与地下城

  • 2019-06-24
  • 题解

有一个 n (n \leq 20) 的骰子，分别写着 0 \cdots (n - 1) ，每个数被投掷出来的概率是相同的，现在我们投掷了 m (m \leq 2\times10^5) 次，造成的伤害为 m 次骰子的点数和 \xi ，求 Pr[a \leq \xi \leq b] 。

Link

Solution 1

对于范围较小的数据，显然我们可以通过多项式卷积来维护转移。设 \displaystyle F(x) = \sum_{i=0}^{n-1} \frac 1n x^i ，G(x) 满足 [x^n]G(x) = [x^{n-1}]G(x) + [x^n]F^m(x) ，则显然 E(\xi) = [x^b] G(x) - [x^{a-1}] G(x) 。由于值域是实数所以只能使用 FFT ，可以获得 60 分的好成绩，时间复杂度 O(nm (\log n + \log m))。

考虑如何进行优化，题目对允许的误差较大，故我们可以舍去一些小于阈值 \epsilon 的系数，易证系数呈单峰状物故剩下的系数是连续的，我们仍然可以用一般的方法进行卷积。打表可以发现当 \epsilon 取 10^{-14} 时，剩余的项数 \leq 2\times10^6 ，设为 L。可以通过此题，时间复杂度为 O(L (\log L + \log m)) 。

Solution 2

设 f(x) 为 \xi 的概率密度函数，F(x) = Pr[\xi \geq x] ，则

E(\xi) = \int_{-\infty}^\infty f(x) x \mathrm dx

对于服从正态分布的随机变量 \xi \sim N(\mu, \sigma^2) ，\displaystyle f(x) = \frac 1 {\sqrt {2 \pi \sigma^2}} e^{-\frac {(x-\mu)^2} {2\sigma^2}} 。显然对于这样的 X ，我们可以方便的用对 f(x) x 进行自适应辛普森积分得到 E(\xi) 。

不幸的是，我们的 \xi 不服从正态分布，甚至 TMD 还是离散的。然鹅由于此题对精度的要求较低，我们可以用一个服从正态分布的概率密度函数来拟合，这就需要用到 中心极限定理 的一个推论。

对于 m 个独立同分布随机变量 x_i ，期望为 \mu ，标准差为 \sigma 。那么设

Y_m = \frac {\sum_{i=1}^m x_i - m \mu} {\sqrt {m \sigma^2}}

在 m 足够大时可以认为 Y_m \sim N(0, 1) 。

对应到此题中，我们可以发现

\sum_{i=1}^m x_i \in [A, B] \Leftrightarrow Y_m \in [\frac {A - m\mu} {\sqrt {m \sigma^2}}, \frac {B - m\mu} {\sqrt {m \sigma^2}}]

显然在 m 足够大时，我们对于 f_{Y_m}(x) x 进行自适应辛普森积分即可。

但 however ，在 m 小到一定程度时，我们并不能认为 Y_m 服从正态分布，暴力处理即可。

Solution 3

这种算法是 Solution 1 的一种延伸，但仅可以用在对质数去模的情况下。

注意到我们需要对 F^m(x) 求一次前缀和，不方便我们的式子推导，我们可以通过乘以 \displaystyle \frac 1 {1 - x} 来代替。也就是说我们需要求 \displaystyle G(x) = \frac {F^m(x)} {1 - x} 的单项系数。

F(x) = \sum_{i=0}^{n-1} \frac 1 n x^i = \frac 1 n \sum_{i=0}^{n-1} x^i = \frac {1 - x^n} {n(1 - x)} G(x) = \frac 1 {1 - x} \left( \frac {1 - x^n} {n(1 - x)}\right) ^y = \frac 1 {n^m} \left( 1 - x^n \right)^m \left( \frac 1 {1-x} \right)^{m+1}

对于求具体的某项直接二项式定理拆开，还是比较套路的。

\begin{aligned} [x^k] G(x) &= \frac 1 {n^m} [x^k] \left( 1 - x^n \right)^m \left( 1-x \right)^{-(m+1)} \\ &= \frac 1 {n^m} [x^k] \left( \sum_{i=0}^m \binom m i (-1)^i x^{in}\right) \left( \sum_{i=0}^m \binom {m + i} m x^i \right) \\ &= \frac 1 {n^m} \sum_{i=0}^{\left\lfloor \frac k n \right\rfloor} (-1)^i \binom m i \binom {m + k - in} {m} \end{aligned}

时间复杂度 O(m) ，非常优秀。

// =================================

#include <bits/stdc++.h>
#define ll long long
#define debug(...) ((void)0)
#ifndef debug
#define debug(...) fprintf(stderr,__VA_ARGS__)
#endif
namespace ringo {
template <class T> inline void read(T &x) {
    x = 0; char c = getchar(); bool f = 0;
    while (!isdigit(c)) f ^= c == '-', c = getchar();
    while (isdigit(c)) x = x * 10 + c - '0', c = getchar();
    if (f) x = -x;
}
template <class T> inline void print(T x) {
    if (x < 0) putchar('-'), x = -x;
    if (x > 9) print(x / 10);
    putchar('0' + x % 10);
}
template <class T> inline void print(T x, char c) { print(x), putchar(c); }
template <class T> inline void print(T a, int l, int r, std::string s = "") {
    if (s != "") std::cout << s << ": ";
    for (int i = l; i <= r; i++) print(a[i], " \n"[i == r]);
}

const double eps = 1e-14, pi = acos(-1);
const int N = 4e6 + 10;
int T, n, m, rev[N];

struct complex {
    double x, y;
    inline complex operator+(const complex &o) { return { x + o.x, y + o.y }; }
    inline complex operator-(const complex &o) { return { x - o.x, y - o.y }; }
    inline complex operator*(const complex &o) { return { x * o.x - y * o.y, x * o.y + y * o.x }; }
} w1[N], w2[N];

struct poly {
    double a[N];
    int n, p;
    inline double get(int k) { k -= p; return k < 0 ? 0 : (k >= n ? a[n - 1] : a[k]); }
    inline void print() { for (int i = 0; i < n; i++) printf("%.2lf%c", a[i], " \n"[i == n - 1]); }
} f, g;

inline void fft(complex *a, int lim, complex *w) {
    for (int i = 0; i < lim; i++) if (i < rev[i]) std::swap(a[i], a[rev[i]]);
    for (int len = 1; len < lim; len <<= 1)
        for (int i = 0; i < lim; i += (len << 1))
            for (int j = 0; j < len; j++) {
                complex x = a[i + j], y = w[j + len] * a[i + j + len];
                a[i + j] = x + y, a[i + j + len] = x - y;
            }
}

inline void mul(poly &f, poly &g, poly &h) {
    static complex a[N], b[N];
    int len = (f.n + g.n - 1), lim = 1, k = 0;
    while (lim < len) lim <<= 1, ++k;
    for (int i = 0; i < lim; i++) rev[i] = (rev[i >> 1] >> 1) | ((i & 1) << (k - 1));
    for (int len = 1; len < lim; len <<= 1) {
        w1[len] = w2[len] = { 1, 0 };
        complex wn1 = { cos(pi / len),  sin(pi / len) };
        complex wn2 = { cos(pi / len), -sin(pi / len) };
        for (int i = 1; i < len; i++) {
            w1[i + len] = w1[i + len - 1] * wn1;
            w2[i + len] = w2[i + len - 1] * wn2;
        }
    }
    // printf(">> mul n=%d m=%d l=%d lim=%d k=%d | %u %u %u \n", f.n, g.n, len, lim, k, &f, &g, &h), f.print(), g.print();
    for (int i = 0; i < lim; i++) {
        a[i] = { i < f.n ? f.a[i] : 0, 0 };
        b[i] = { i < g.n ? g.a[i] : 0, 0 };
    }
    fft(a, lim, w1), fft(b, lim, w1);
    for (int i = 0; i < lim; i++) a[i] = a[i] * b[i];
    fft(a, lim, w2);
    int flag = false, first = 0, second = len;
    for (int i = 0; i < lim; i++) {
        a[i].x /= lim;
        if (!flag && a[i].x > eps) flag = true, first = i;
        if (flag && a[i].x <= eps) flag = false, second = i;
    }
    h.n = second - first, h.p = f.p + g.p + first;
    for (int i = 0; i < h.n; i++) h.a[i] = a[i + first].x;
    // h.print();
}

inline void fpow(poly &a, int b, poly &s) {
    s.n = s.a[0] = 1, s.p = 0;
    for (; b; b >>= 1, mul(a, a, a))
        if (b & 1) mul(s, a, s);
}

void main() {
    for (read(T); T--; ) {
        read(n), read(m);
        // memset(f.a, 0, sizeof(f.a));
        // memset(g.a, 0, sizeof(g.a)); 
        f.n = n, f.p = 0; for (int i = 0; i < n; i++) f.a[i] = 1.0 / n;
        // f.print();
        fpow(f, m, g);
        // g.print();
        for (int i = 1; i <= (n - 1) * m; i++) g.a[i] += g.a[i - 1];
        for (int i = 1, a, b; i <= 10; i++) read(a), read(b), printf("%.6lf\n", g.get(b) - g.get(a - 1));
    }
}

} signed main() {
#ifdef MEMSET0_LOCAL_ENVIRONMENT
    freopen("1.in", "r", stdin);
#endif
    return ringo::main(), 0;
}
