洛谷5334 [JSOI2019]节日庆典

  • 2019-05-06
  • 题解

给定字符串 S，对于每个前缀求出它的最小表示，如果有多个求起始位置下标最小的那个。|S| \leq 3 \times 10^6。

一道非常妙的串串题。

考虑依次计算每一个前缀 S[1:L] 的答案，定义 s_i = S[i:L]。

维护一个集合 E 使得答案会且仅会从中选择。

考虑对于两个数 i, j，如果 \operatorname{lcp}(s_i, s_j) < L - j + 1，那么我们可以比较出 i 的表示和 j 的表示的大小，就可以确定较大的一个肯定不能放入 E 中。由于每次 L 只会 +1，所以可以不用重新求 \operatorname{lcp}，只要判断下新加的那位即可。

然而这样并不足够优秀，需要考虑一个优化：对于 s_i 和 s_j 满足 i, j \in E 且 \operatorname{len}(s_j) < \operatorname{len}(s_i) < 2 \cdot \operatorname{len}(s_j)，显然因为 i, j \in E，所以 s_j 是 s_i 的一个前缀，再根据后缀本质的性质，得到 s_j 是 s_i 的一个后缀，也就是说 s_i 有一个长度为 \operatorname{len} (s_j) 的 border。然而由于 \operatorname{len}(s_j) < \operatorname{len}(s_i) < 2 \cdot \operatorname{len}(s_j)，所以 s_i 还有一个长度更短的 border，也就是 s_{2j-i}，设为 s_k。

考虑添加了一个字符 c 比较 s_i + c, s_j + c, s_k + c 的大小关系。定义 \operatorname{len} (s_j) = R，那么易得 S[i : i + R - 1] = S[j : i + R - 1] = s[k : k + R - 1]，令其为 T。

  • 若 c = T[0] 跳过不管。
  • 若 c < T[0]，因为 s_i = s_j + s_k = s_j + T + (s_k - T), s_j = s_k + T，所以 s_i < s_j < s_k。
  • 若 c > T[0]，同理 s_i > s_j > s_k。

故无论如何 j 不可能作为答案，可以从 E 中舍去。（注意这样的 k 必定存在）

考虑如何根据 E 求答案，在 E 中依次扫过去，判断的时候分别根据 \operatorname{lcp}(s_1, s_{j-i+1}) 和 \operatorname{lcp}(s_1, s_{i-j+L+1}) 处理即可，这里不多废话。

发现我们只需要求 \operatorname{lcp}(s_1, s_k), k \in [1, n]，考虑用 Z-Algorithm 代替朴素后缀数组 / 后缀自动机来求 LCP。

可以证明对于这样的 E 满足 |E| \leq \log n，总时间复杂度为 \mathcal O(n \times |E|^2) = \mathcal O(n \log^2 n)。

// =================================

//   website: https://memset0.cn/

#include <bits/stdc++.h>
#define ll long long
#define debug(...) ((void)0)
#ifndef debug
#define debug(...) fprintf(stderr,__VA_ARGS__)
#endif
namespace ringo {
inline char read() {
    static const int IN_LEN = 1000000; static char buf[IN_LEN], *s, *t;
    return (s == t ? t = (s = buf) + fread(buf, 1, IN_LEN, stdin), (s == t ? -1 : *s++) : *s++);
}
template <class T> inline void read(T &x) {
    static bool f; static char c;
    for (f = 0, c = read(); !isdigit(c); c = read()) { f ^= c == '-'; if (c == -1) return; }
    for (x = 0; isdigit(c); c = read()) x = ((x + (x << 2)) << 1) + (c ^ '0');
    if (f) x = -x;
}
const int OUT_LEN = 10000000; char obuf[OUT_LEN], *ooh = obuf;
inline void print(char c) {
    if (ooh == obuf + OUT_LEN) fwrite(obuf, 1, OUT_LEN, stdout), ooh = obuf;
    *ooh++ = c;
}
template<class T> inline void print(T x) {
    static int buf[30], cnt;
    if (!x) { print('0'); return; }
    if (x < 0) print('-'), x = -x;
    for (cnt = 0; x; x /= 10) buf[++cnt] = x % 10 + '0';
    while (cnt) print((char)buf[cnt--]);
}
inline void flush() { fwrite(obuf, 1, ooh - obuf, stdout); }
template <class T> inline void print(T x, char c) { print(x), print(c); }
template <class T> inline void print(T a, int l, int r) {
    for (int i = l; i <= r; i++) print(a[i], " \n"[i == r]);
}

const int N = 3e6 + 10, M = 23;
char s[N]; bool del[N];
int n, siz, len, ans[N], pre[N], tmp[N];

struct setClass {
    int top, stk[100];
    inline void clear() { top = 0; }
    inline void pop_back() { top--; }
    inline int size() { return top; }
    inline void push_back(int x) { stk[top++] = x; }
    inline int& operator[](size_t k) { return stk[k]; }
    inline void reset() {
        int siz = 0;
        for (int i = 0; i < top; i++) {
            if (del[i]) del[i] = 0;
            else tmp[siz++] = stk[i];
        }
        top = siz;
        for (int i = 0; i < top; i++)
            stk[i] = tmp[i];
    }
} set, lis;

void zAlgorithm() {
#define z pre
    for (int i = 2, l = 1, r = 1; i <= n; i++) {
        if (i <= r) z[i] = std::min(r - i + 1, z[i - l + 1]);
        while (i + z[i] <= n && s[1 + z[i]] == s[i + z[i]]) ++z[i];
        if (i + z[i] - 1 > r) l = i, r = i + z[i] - 1;
    }
    z[1] = n;
    // for (int i = 1; i <= n; i++) print(z[i], " \n"[i == n]);
#undef z
}

inline int better(int i, int j, int L) {
    int lcp = pre[i - j + L + 1 ];
    if (lcp < j - i) {
        if (s[i - j + L + 1 + lcp] < s[1 + lcp]) return i;
        if (s[i - j + L + 1 + lcp] > s[1 + lcp]) return j;
    }
    lcp = pre[1 + j - i];
    if (lcp < i - 1) {
        if (s[1 + lcp] < s[1 + j - i + lcp]) return i;
        if (s[1 + lcp] > s[1 + j - i + lcp]) return j;
    }
    return i;
}

void main() {
    scanf("%s", s + 1), n = len = strlen(s + 1);
    zAlgorithm(), set.push_back(1), ans[1] = 1;
    for (int L = 2; L <= len; L++) {
        lis = set, lis.push_back(L), set.clear();
        for (int j = 0; j < lis.size(); j++) {
            bool flag = 1;
            for (int i = set.size() - 1; i >= 0; i--) {
                if (s[set[i] + L - lis[j]] < s[lis[j] + L - lis[j]]) { flag = 0; break; }
                if (s[set[i] + L - lis[j]] > s[lis[j] + L - lis[j]]) set.pop_back();
            }
            if (flag) set.push_back(lis[j]);
        }
        set.reset();
        // for (int i = 0; i < set.size(); i++) std::cerr << set[i] << " \n"[i == set.size() - 1];
        for (int q = 0; q < set.size(); q++)
            for (int p = 0; p < q; p++) {
                int i = set[p], j = set[q];
                if (L - j + 1 < L - i + 1 && L - i + 1 < 2 * (L - j + 1)) {
                    del[q] = 1;
                    break;
                }
            }
        set.reset();
        // for (int i = 0; i < set.size(); i++) std::cerr << set[i] << " \n"[i == set.size() - 1];
        ans[L] = set[0];
        for (int i = 1; i < set.size(); i++) {
            ans[L] = better(ans[L], set[i], L);
        }
    }
    print(ans, 1, len);
}

} signed main() {

    return 0;
}
