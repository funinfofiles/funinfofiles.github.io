SPOJ2002 Random Number Generator

  • 2019-06-24
  • 题解

给定一个 n (n \leq 10) 个整数 a_i (0 \leq a_i \leq 10) ，连续随机变量 \xi_i 在 [-a_i, a_i] 的范围内随机生成，定义 \displaystyle S = \sum_{i=1}^n \xi_i ，求 Pr[A \leq S \leq B] (-100 \leq A, B \leq 100) 。

Vjudge Link | SPOJ Link

可以发现 \xi_i : [-a_i, a_i] 处理起来较为复杂，我们可以将区间整体平移 a_i ，即令 \displaystyle a'_i = 2 a_i, A' = A + \sum_{i=1}^n a_i, B' = B + \sum_{i=1}^n a_i 。题意转变为 \xi_i : [0, a'_i] ，求 \displaystyle Pr[A' \leq \sum_{i=1}^n \xi_i \leq B'] 。下文中我们直接用 a 指代 a' ，以此类推。同时，我们可以把 Pr[A \leq S \leq B] 拆为 Pr[S \leq B] - Pr[S \leq A] 此题中我们需要多次用到类似的思路。

定义多面体

A(a) = \int_0^{a_1} \int_0^{a_2} \cdots \int_0^{a_n} 1 \mathrm dx_1 \mathrm dx_2 \cdots \mathrm dx_n B(a) = \int_{a_1}^{+\infty} \int_{a_2}^{+\infty} \cdots \int_{a_n}^{+\infty} 1 \mathrm dx_1 \mathrm dx_2 \cdots \mathrm dx_n T(x) = \int_0^{+\infty} \int_0^{+\infty} \cdots \int_0^{+\infty} \left[\sum_{i=1}^n x_i \leq K \right] \mathrm dx_1 \mathrm dx_2 \cdots \mathrm dx_n

相当于我们需要求 A(a) 与 T(K \in \{A, B\}) 的高维体积交，记为 A(a) \cup T(K)。

这不好算，考虑容斥，我们硬点一个点集 S \subseteq [1, n] ，定义 b_{S, i} = \left\{ \begin{aligned} &a_i &(x \in S) \\ &0 & (x \not\in S)\end{aligned} \right. ，则对 A(a) 与 T(K) 的体积并的贡献等于多面体 B(b) 和 T 的高维体积并乘以 (-1)^{|S|} 。显然 B(b) \cup T(K) = B(0) \cup T(K - \sum b_i) ，而 \displaystyle B(0) \cup T(x) = \frac {x^n} {n!} ，于是我们就可以通过枚举集合 S 来快速计算了。

时间复杂度 O(T n 2^n) 。

注意此题卡精度。

// =================================
//   author: memset0
//   date: 2019.06.24 14:31:07
//   website: https://memset0.cn/
// =================================
#include <bits/stdc++.h>
#define ll long long
#define debug(...) ((void)0)
#ifndef debug
#define debug(...) fprintf(stderr,__VA_ARGS__)
#endif
namespace ringo {
template <class T> inline void read(T &x) {
    x = 0; char c = getchar(); bool f = 0;
    while (!isdigit(c)) f ^= c == '-', c = getchar();
    while (isdigit(c)) x = x * 10 + c - '0', c = getchar();
    if (f) x = -x;
}
template <class T> inline void print(T x) {
    if (x < 0) putchar('-'), x = -x;
    if (x > 9) print(x / 10);
    putchar('0' + x % 10);
}
template <class T> inline void print(T x, char c) { print(x), putchar(c); }
template <class T> inline void print(T a, int l, int r, std::string s = "") {
    if (s != "") std::cout << s << ": ";
    for (int i = l; i <= r; i++) print(a[i], " \n"[i == r]);
}

const int N = 15;
ll tmp;
long double ans;
int n, T, A, B, a[N];

inline long double get(int x) {
    if (x < 0) return 0;
    long double res = 1;
    for (int i = 1; i <= n; i++)
        res *= x / (long double)i;
    return res;
}

void main() {
    for (read(T); T--; ) {
        tmp = 1, ans = 0;
        read(n), read(A), read(B);
        for (int i = 1; i <= n; i++) {
            read(a[i]);
            A += a[i], B += a[i], a[i] <<= 1, tmp *= a[i];
        }
        for (int S = 0; S < (1 << n); S++) {
            int sum = 0, cnt = 0;
            for (int i = 1; i <= n; i++)
                if ((S >> (i - 1)) & 1)
                    sum += a[i], ++cnt;
            // printf(">> %d >> %d >> %.2lf %.2lf\n", S, sum, get(B - sum), get(A - sum));
            ans += (cnt & 1 ? -1 : 1) * (get(B - sum) - get(A - sum));
        }
        printf("%.9Lf\n", ans / tmp);
    }
}

} signed main() {
#ifdef MEMSET0_LOCAL_ENVIRONMENT
    freopen("1.in", "r", stdin);
    // freopen("1.out", "w", stdout) ;
#endif
    return ringo::main(), 0;
}
